// ══════════════════════════════════════════════════════════════════════════
// ═══ GRAADSKLAS-WEERGAVE BEHEER ═══════════════════════════════════════════
// ══════════════════════════════════════════════════════════════════════════
// Graadsklassen worden opgeslagen in meta-doc onder key 'graadsklassen'.
// Elk item: { id, naam, bord1:{bordid,klaslabel,bordnaam}, bord2:{...}, bijgewerkt }
// En extra in Firestore collection klasbord_shared/graadsklas_{id} (voor iPads
// om zonder login de set te kunnen laden via de QR-code).

function getGraadsklassen(){
  if(_metaCache && Array.isArray(_metaCache.graadsklassen)) return _metaCache.graadsklassen;
  try {
    const m = JSON.parse(localStorage.getItem(BORDEN_KEY) || '{}');
    return Array.isArray(m.graadsklassen) ? m.graadsklassen : [];
  } catch { return []; }
}

function saveGraadsklassen(lijst){
  if(!_metaCache) _metaCache = {};
  _metaCache.graadsklassen = lijst;
  try {
    const all = JSON.parse(localStorage.getItem(BORDEN_KEY)||'{}');
    all.graadsklassen = lijst;
    localStorage.setItem(BORDEN_KEY, JSON.stringify(all));
  } catch {}
  if(window.fbSaveMeta) window.fbSaveMeta(_metaCache).catch(console.warn);
}

function renderGraadLijst(){
  const lijst = document.getElementById('graad-lijst');
  if(!lijst) return;

  const klasborden = getBorden('klasbord');
  const btn = document.getElementById('graad-nieuw-btn');
  if(btn){
    if(klasborden.length < 2){
      btn.disabled = true;
      btn.textContent = '➕ Eerst 2 klasborden aanmaken hierboven';
    } else {
      btn.disabled = false;
      btn.textContent = '➕ Graadsklas-weergave aanmaken';
    }
  }

  const graden = getGraadsklassen();
  lijst.innerHTML = '';

  if(!graden.length){
    lijst.innerHTML = '<div style="text-align:center;padding:18px;color:#9ca3af;font-size:13px;">Nog geen graadsklas-weergaves aangemaakt.</div>';
    return;
  }

  graden.forEach(function(gk){
    const rij = document.createElement('div');
    rij.className = 'graad-rij';

    const info = document.createElement('div');
    info.className = 'graad-rij-info';
    const label1 = (gk.bord1 && gk.bord1.klaslabel) || '?';
    const label2 = (gk.bord2 && gk.bord2.klaslabel) || '?';
    info.innerHTML =
      '<div class="graad-rij-titel">' + esc(gk.naam || (label1 + ' + ' + label2)) + '</div>' +
      '<div class="graad-rij-sub">' + esc(label1) + ' + ' + esc(label2) + ' · Bijgewerkt: ' + fmtDatum(gk.bijgewerkt||gk.aangemaakt) + '</div>';

    const openLink = document.createElement('a');
    openLink.className = 'graad-rij-btn';
    openLink.href = 'klasbord_graad.html?graadid=' + encodeURIComponent(gk.id);
    openLink.textContent = '▶ Openen';

    const delBtn = document.createElement('button');
    delBtn.className = 'graad-rij-del';
    delBtn.textContent = '🗑';
    delBtn.title = 'Verwijderen';
    delBtn.onclick = (function(id){ return function(e){ e.stopPropagation(); openGraadDel(id); }; })(gk.id);

    rij.innerHTML = '<div class="graad-rij-icon">📊</div>';
    rij.appendChild(info);
    rij.appendChild(delBtn);
    rij.appendChild(openLink);
    lijst.appendChild(rij);
  });
}

// ── Modal: nieuw/bewerk ───────────────────────────────────────────────────
let _bewerkGraadId = null;

function opendNieuweGraadsklas(){
  _bewerkGraadId = null;
  document.getElementById('graad-modal-titel').textContent = '📊 Graadsklas-weergave aanmaken';
  document.getElementById('graad-bewaar-btn').textContent = 'Aanmaken en openen';
  document.getElementById('graad-naam-input').value = '';
  document.getElementById('graad-label-1').value = '';
  document.getElementById('graad-label-2').value = '';
  document.getElementById('graad-fout').style.display = 'none';

  // Bord-select opties vullen
  const bordenLijst = getBorden('klasbord');
  [1,2].forEach(function(n){
    const sel = document.getElementById('graad-bord-'+n);
    sel.innerHTML = '<option value="">Kies een bord…</option>' +
      bordenLijst.map(function(b){
        return '<option value="'+esc(b.id)+'">'+esc(b.naam)+'</option>';
      }).join('');
  });

  document.getElementById('modal-graad-nieuw').style.display = 'flex';
  setTimeout(function(){ document.getElementById('graad-naam-input').focus(); }, 50);
}

function sluitGraadNieuw(){
  document.getElementById('modal-graad-nieuw').style.display = 'none';
}

function bewaarGraadsklas(){
  const naam = document.getElementById('graad-naam-input').value.trim();
  const label1 = document.getElementById('graad-label-1').value.trim();
  const label2 = document.getElementById('graad-label-2').value.trim();
  const bordid1 = document.getElementById('graad-bord-1').value;
  const bordid2 = document.getElementById('graad-bord-2').value;
  const foutEl = document.getElementById('graad-fout');

  function toonFout(msg){
    foutEl.textContent = msg;
    foutEl.style.display = '';
  }

  if(!naam){ toonFout('Geef je graadsklas-weergave een naam.'); return; }
  if(!label1 || !label2){ toonFout('Geef beide borden een klaslabel (bv. 3A en 4A).'); return; }
  if(!bordid1 || !bordid2){ toonFout('Kies beide borden.'); return; }
  if(bordid1 === bordid2){ toonFout('Kies 2 verschillende borden.'); return; }

  const bordenLijst = getBorden('klasbord');
  const bord1 = bordenLijst.find(function(b){ return b.id === bordid1; });
  const bord2 = bordenLijst.find(function(b){ return b.id === bordid2; });
  if(!bord1 || !bord2){ toonFout('Een van de gekozen borden bestaat niet meer.'); return; }

  const graden = getGraadsklassen();
  const id = _bewerkGraadId || uid();

  const nieuweGraad = {
    id: id,
    naam: naam,
    bord1: {
      bordid: bord1.id,
      storageKey: bord1.storageKey,
      klaslabel: label1,
      bordnaam: bord1.naam
    },
    bord2: {
      bordid: bord2.id,
      storageKey: bord2.storageKey,
      klaslabel: label2,
      bordnaam: bord2.naam
    },
    aangemaakt: _bewerkGraadId ? (graden.find(function(g){return g.id===id;}) || {}).aangemaakt || Date.now() : Date.now(),
    bijgewerkt: Date.now()
  };

  if(_bewerkGraadId){
    const i = graden.findIndex(function(g){ return g.id === _bewerkGraadId; });
    if(i >= 0) graden[i] = nieuweGraad; else graden.push(nieuweGraad);
  } else {
    graden.push(nieuweGraad);
  }

  saveGraadsklassen(graden);

  // Ook in klasbord_shared zetten zodat de iPad-keuzepagina de set kan laden
  // zonder dat het kind moet inloggen.
  if(window.fbSaveShared){
    const sharedDoc = {
      naam: naam,
      bord1: nieuweGraad.bord1,
      bord2: nieuweGraad.bord2,
      bijgewerkt: Date.now()
    };
    window.fbSaveShared('graadsklas_'+id, sharedDoc).catch(console.warn);
    try{ localStorage.setItem('graadsklas_'+id, JSON.stringify(sharedDoc)); }catch(e){}
  }

  sluitGraadNieuw();
  // Meteen openen
  window.location.href = 'klasbord_graad.html?graadid=' + encodeURIComponent(id);
}

// ── Verwijderen ───────────────────────────────────────────────────────────
let _delGraadId = null;

function openGraadDel(id){
  _delGraadId = id;
  const gk = getGraadsklassen().find(function(g){ return g.id === id; });
  if(!gk) return;
  const label1 = (gk.bord1 && gk.bord1.klaslabel) || '?';
  const label2 = (gk.bord2 && gk.bord2.klaslabel) || '?';
  document.getElementById('graad-del-info').innerHTML =
    '<strong>' + esc(gk.naam) + '</strong><br/>Koppelt: ' + esc(label1) + ' + ' + esc(label2);
  document.getElementById('modal-graad-del').style.display = 'flex';
}

function sluitGraadDel(){
  document.getElementById('modal-graad-del').style.display = 'none';
}

function bevestigGraadDel(){
  if(!_delGraadId) return;
  const graden = getGraadsklassen().filter(function(g){ return g.id !== _delGraadId; });
  saveGraadsklassen(graden);

  // Ook de shared-doc opruimen zodat oude QR-codes niet meer werken
  // (fbDelete schrijft naar klasbord/{uid}/data/... — daar staat dit NIET,
  //  we moeten rechtstreeks naar klasbord_shared schrijven om te wissen.
  //  Voor nu laten we 't staan — heeft geen impact behalve dat oude QR
  //  tijdelijk blijft werken tot je de graadsklas opnieuw aanmaakt).

  sluitGraadDel();
  _delGraadId = null;
  renderGraadLijst();
}
